app.post("/add-keywords", async (req, res) => {
    try {
        const { keywordInput } = req.body

        if (!keywordInput) {
            return res.status(400).json({ success: false, message: "No keywords provided" })
        }

        const keywordList = keywordInput
            .split(/,|\n/)
            .map(k => k.trim())
            .filter(Boolean)

        if (keywordList.length === 0) {
            return res.status(400).json({ success: false, message: "Empty keywords" })
        }

        const values = keywordList.map(k => ({ keyword: k }))
        await db.insert(keywords).values(values)

        res.json({ success: true, message: `Added ${keywordList.length} keyword(s)` })

    } catch (err) {
        console.error("ADD KEYWORDS ERROR:", err)
        res.status(500).json({ success: false, message: "Error adding keywords: " + err.message })
    }
})

app.post("/run-keyword-scraper", async (req, res) => {
    try {
        const keywordsList = await db.select().from(keywords)

        if (keywordsList.length === 0) {
            return res.json({ success: false, message: "No keywords saved in database" })
        }

        res.json({ success: true, message: `Started scraping ${keywordsList.length} keyword(s)...` })

        ;(async () => {
            for (let k of keywordsList) {
                if (!k.keyword) continue
                const parts = k.keyword.split(" in ")
                const keywordPart = parts[0]
                const locationPart = parts[1]
                if (!keywordPart || !locationPart) {
                    console.log(`Skipping "${k.keyword}" - format must be "Term in City"`)
                    continue
                }
                console.log(`Scraping: ${k.keyword}`)
                await scrapeBusinesses(keywordPart, [locationPart])
            }
            console.log("All keyword scraping done")
        })()

    } catch (err) {
        console.error("SCRAPER ERROR:", err)
        res.status(500).json({ success: false, message: "Error: " + err.message })
    }
})

app.post("/run-selected-keywords", async (req, res) => {
    try {
        const { keywords: selectedKeywords } = req.body

        if (!Array.isArray(selectedKeywords) || selectedKeywords.length === 0) {
            return res.json({ success: false, message: "No keywords provided" })
        }

        const valid = []
        const skipped = []

        for (const kw of selectedKeywords) {
            const parts = kw.split(" in ")
            if (!parts[0] || !parts[1]) {
                skipped.push(kw)
            } else {
                valid.push({ keyword: kw, keywordPart: parts[0].trim(), locationPart: parts[1].trim() })
            }
        }

        if (valid.length === 0) {
            return res.json({ success: false, message: 'All keywords have invalid format. Must be "Term in City"' })
        }

        res.json({
            success: true,
            message: `Started scraping ${valid.length} keyword(s)...` + (skipped.length ? ` Skipped: ${skipped.join(", ")}` : "")
        })

        ;(async () => {
            for (const item of valid) {
                console.log(`Scraping: ${item.keyword}`)
                await scrapeBusinesses(item.keywordPart, [item.locationPart])
                console.log(`Done: ${item.keyword}`)
            }
            console.log("Selected keyword scraping done")
        })()

    } catch (err) {
        res.status(500).json({ success: false, message: err.message })
    }
})

app.post("/run-manual-scrape", async (req, res) => {
    try {
        const { keyword, location } = req.body
        if (!keyword || !location) {
            return res.json({ success: false, message: "Keyword and location are required" })
        }
        const locations = location.split(",").map(l => l.trim()).filter(Boolean)

        res.json({ success: true, message: `Scraping "${keyword}" in ${locations.join(", ")}...` })

        ;(async () => {
            console.log(`Manual scrape: ${keyword} in ${locations.join(", ")}`)
            await scrapeBusinesses(keyword, locations)
            console.log(`Manual scrape done`)
        })()

    } catch (err) {
        res.status(500).json({ success: false, message: err.message })
    }
})

app.get("/export-leads", async (req, res) => {
  try {
    const { city } = req.query

    let query = db.select().from(leads)

    if (city) {
      query = query.where(eq(leads.city, city))
    }

    const allLeads = await query

    if (!allLeads.length) {
      return res.status(404).send("No leads found")
    }

    const workbook = new ExcelJS.Workbook()
    const sheet = workbook.addWorksheet("Leads")

    // Columns
    sheet.columns = [
      { header: "Name", key: "name", width: 30 },
      { header: "Keyword", key: "keyword", width: 25 },
      { header: "City", key: "city", width: 20 },
      { header: "Phone", key: "phone", width: 20 },
      { header: "Website", key: "website", width: 30 }
    ]

    // Add rows
    allLeads.forEach(l => {
      sheet.addRow({
        name: l.name || "",
        keyword: l.keyword || "",
        city: l.city || "",
        phone: l.phone || "",
        website: l.website || ""
      })
    })

    // Styling (optional but nice)
    sheet.getRow(1).font = { bold: true }

    res.setHeader(
      "Content-Type",
      "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"
    )

    const fileName = city
      ? `leads_${city}.xlsx`
      : `all_leads.xlsx`

    res.setHeader(
      "Content-Disposition",
      `attachment; filename=${fileName}`
    )

    await workbook.xlsx.write(res)
    res.end()

  } catch (err) {
    console.error("EXPORT ERROR:", err)
    res.status(500).send("Error exporting leads")
  }
})

      bar.textContent = msg;
      bar.style.display = 'block';

    // ── Poll until lead count grows, then reload ──
    // baselineCount: the count before scraping started
    // labelText: shown in the loader while polling
    // timeoutMs: give up after this many ms (default 5 min)
    function pollUntilNewData(baselineCount, labelText, timeoutMs) {
      var deadline = Date.now() + (timeoutMs || 300000); // 5 min max
      var dots = 0;

      function tick() {
        dots = (dots + 1) % 4;
        var dotStr = '.'.repeat(dots + 1);
        updateLoaderText((labelText || 'Scraping') + dotStr);

        if (Date.now() > deadline) {
          hideLoader();
          showStatus('Scraping timed out. Reloading to show latest data.', true);
          setTimeout(function(){ location.reload(); }, 1500);
          return;
        }

        fetch(BASE + '/leads-count')
          .then(function(r){ return r.json(); })
          .then(function(data) {
            if (data.count > baselineCount) {
              // New records found — reload page to show them
              updateLoaderText('Loading new data...');
              location.reload();
            } else {
              // Not yet — check again in 2s
              setTimeout(tick, 2000);
            }
          })
          .catch(function() {
            // Network blip — retry
            setTimeout(tick, 3000);
          });
      }

      setTimeout(tick, 2000); // first check after 2s
    }

    // ── Get current lead count then run a scrape action ──
    // action: function(baselineCount) — should start the scrape, returns a Promise<{success, message}>
    // loaderText: text shown while scraping
    function scrapeWithPolling(action, loaderText) {
      showLoader('Starting...');
      // Step 1: snapshot current count
      fetch(BASE + '/leads-count')
        .then(function(r){ return r.json(); })
        .then(function(countData) {

          return action().then(function(result) {
            if (!result.success) {
              hideLoader();
              showStatus('Error: ' + result.message, true);
              return;
            }
            // Step 3: poll until count grows
            showLoader(loaderText || 'Scraping');
            pollUntilNewData(baseline, loaderText || 'Scraping');
          });
        })
        .catch(function(e) {
          hideLoader();
          showStatus('Network error: ' + e.message, true);
        });
    }

      if (!confirm('Delete ' + checked.length + ' keyword(s)?')) return;
      var i = 0;
      function deleteNext() {
        if (i >= checked.length) { showStatus('Deleted ' + checked.length + ' keyword(s).'); loadKeywords(); return; }
        fetch(BASE + '/delete-keyword/' + checked[i].dataset.id, { method: 'DELETE' })
          .then(function(r){ return r.json(); })
          .then(function(){ i++; deleteNext(); })
          .catch(function(e){ showStatus('Error deleting: ' + e.message, true); });
      }
      deleteNext();
    }

app.get("/", async (req, res) => {
  const keyword = req.query.keyword
  const locationInput = req.query.location
  const filterCity = req.query.filterCity

  const page = parseInt(req.query.page) || 1
  const limit = 10

  try {
    const citiesResult = await db.select({ city: leads.city }).from(leads)
    const citiesList = [...new Set(citiesResult.map(c => c.city).filter(Boolean))]

    let query = db.select().from(leads)
    let countQuery = db.select({ count: sql`count(*)` }).from(leads)

    if (filterCity) {
      query = query.where(eq(leads.city, filterCity))
      countQuery = countQuery.where(eq(leads.city, filterCity))
    }

    const totalResult = await countQuery
    const totalLeads = Number(totalResult[0].count)
    const totalPages = Math.ceil(totalLeads / limit) || 1

    const result = await query.limit(limit).offset((page - 1) * limit)

    res.send(renderHTML(result, totalLeads, page, totalPages, keyword, locationInput, citiesList, filterCity))

  } catch (err) {
    console.error(err)
    res.send(`<h3>Error</h3><pre>${err.message}</pre>`)
  }
})
